# Liuyun Windows MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the first Windows desktop MVP foundation for Liuyun with a Tauri + React + TypeScript app, tested repeat/reminder domain logic, local-only storage boundaries, and a UI matching the approved prototype direction.

**Architecture:** The frontend owns the interactive UI and testable TypeScript domain logic for todo repeat/reminder behavior. Tauri provides the Windows desktop shell and later Rust commands for SQLite, startup, file access, and reminder windows. This first implementation creates a working scaffold and verifies that the Windows development stack can run and build.

**Tech Stack:** Tauri 2, React, TypeScript, Vite, Vitest, CSS variables, Rust stable MSVC toolchain.

import type { RepeatRule } from "./types";

const ISO_WITH_OFFSET = /^(.+)([+-]\d{2}:\d{2}|Z)$/;

function splitIsoOffset(value: string): { body: string; offset: string } {
  const match = value.match(ISO_WITH_OFFSET);
  if (!match) {
    throw new Error(`Expected ISO datetime with timezone offset: ${value}`);
  }
  return { body: match[1], offset: match[2] };
}

function pad(value: number): string {
  return String(value).padStart(2, "0");
}

function daysInMonth(year: number, monthIndex: number): number {
  return new Date(year, monthIndex + 1, 0).getDate();
}

function formatLocalWithOffset(date: Date, offset: string): string {
  return [
    date.getFullYear(),
    "-",
    pad(date.getMonth() + 1),
    "-",
    pad(date.getDate()),
    "T",
    pad(date.getHours()),
    ":",
    pad(date.getMinutes()),
    ":",
    pad(date.getSeconds()),
    offset,
  ].join("");
}

export function computeNextOccurrence(dueAt: string, rule: RepeatRule): string | null {
  if (rule.type === "none") {
    return null;
  }

  const { offset } = splitIsoOffset(dueAt);
  const date = new Date(dueAt);

  if (Number.isNaN(date.getTime())) {
    throw new Error(`Invalid dueAt datetime: ${dueAt}`);
  }

  if (rule.type === "daily") {
    date.setDate(date.getDate() + 1);
    return formatLocalWithOffset(date, offset);
  }

  if (rule.type === "weekly") {
    date.setDate(date.getDate() + 7);
    return formatLocalWithOffset(date, offset);
  }

  const originalDay = date.getDate();
  const nextMonthIndex = date.getMonth() + 1;
  const nextYear = date.getFullYear() + Math.floor(nextMonthIndex / 12);
  const normalizedMonthIndex = nextMonthIndex % 12;
  const clampedDay = Math.min(originalDay, daysInMonth(nextYear, normalizedMonthIndex));
  date.setFullYear(nextYear, normalizedMonthIndex, clampedDay);
  return formatLocalWithOffset(date, offset);
}
```

import type { Todo } from "./types";

const ISO_WITH_OFFSET = /^(.+)([+-]\d{2}:\d{2}|Z)$/;

function splitIsoOffset(value: string): { offset: string } {
  const match = value.match(ISO_WITH_OFFSET);
  if (!match) {
    throw new Error(`Expected ISO datetime with timezone offset: ${value}`);
  }
  return { offset: match[2] };
}

function pad(value: number): string {
  return String(value).padStart(2, "0");
}

function formatLocalWithOffset(date: Date, offset: string): string {
  return [
    date.getFullYear(),
    "-",
    pad(date.getMonth() + 1),
    "-",
    pad(date.getDate()),
    "T",
    pad(date.getHours()),
    ":",
    pad(date.getMinutes()),
    ":",
    pad(date.getSeconds()),
    offset,
  ].join("");
}

export function findDueReminders(todos: Todo[], now: string): Todo[] {
  const nowTime = new Date(now).getTime();

  return todos.filter((todo) => {
    if (todo.status === "done" || !todo.reminderEnabled || !todo.nextReminderAt) {
      return false;
    }
    if (todo.lastRemindedAt === todo.nextReminderAt) {
      return false;
    }
    return new Date(todo.nextReminderAt).getTime() <= nowTime;
  });
}

export function snoozeTodo(todo: Todo, now: string, minutes: number): Todo {
  const { offset } = splitIsoOffset(now);
  const next = new Date(now);
  next.setMinutes(next.getMinutes() + minutes);

  return {
    ...todo,
    nextReminderAt: formatLocalWithOffset(next, offset),
    updatedAt: now,
  };
}
```
